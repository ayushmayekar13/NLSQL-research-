Generated SQL
SELECT
  SUM(p.amount) AS total_revenue
FROM   orders o
JOIN   payments p ON o.order_id = p.order_id
WHERE
  p.status       = 'completed'
  AND p.payment_date >= DATE_TRUNC('month', CURRENT_DATE - INTERVAL '1 month')
  AND p.payment_date <  DATE_TRUNC('month', CURRENT_DATE);
